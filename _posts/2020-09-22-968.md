~~~C++
class Solution {
public:
    int sum = 0;
    int dfs(TreeNode *root){
        if(root == nullptr)
            return 1;
        int rl = dfs(root->left);
        int rr = dfs(root->right);
        if(rl == 0 || rr == 0){
            sum++;
            return 2;
        }
        if(rl == 1 && rr == 1)
            return 0;
        if(rl + rr >= 3)
            return 1;
        return 1;
    }
    int minCameraCover(TreeNode* root) {
        if(root != nullptr){
            if(dfs(root) == 0)
                sum++;
        }
        return sum;
    }
};
~~~
