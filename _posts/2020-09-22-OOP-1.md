1. 求2个或3个正整数中的最大数，用带有默认参数的函数实现。

~~~C++
#include <iostream>
using namespace std;

int Max(int a = 0, int b = 0, int c = 0){//返回a,b,c的最大值
    if(a >= b && a >= c)
        return a;
    else if(b >= a && b >= c)
        return b;
    else
        return c;
}

int main(void){
    int num1,num2,num3;
    cout << "输入两个数的情况:" << endl;
    cin >> num1 >> num2;
    cout << "最大值为" << Max(num1,num2) << endl;
    cout << "输入三个数的情况:" << endl;
    cin >> num1 >> num2 >> num3;    
    cout << "最大值为" << Max(num1,num2,num3) << endl;
    return 0;
}
~~~

   6. 声明一个Circle类，有数据成员Radius（半径）、成员函数GetArea（）计算圆的面积，构造一个Circle的对象进行测试。

      ~~~C++

      using namespace std;
      
      
      const double pi = 3.14;
      class Circle{//定义Circle类
          double Radius;
          public:
              Circle(double r){
                  Radius = r;
              }
              double GetArea(){
                  return Radius*Radius*pi;
              }
      };
      
      int main(void){
          double r;
          cout << "输入半径" << endl;
          cin >> r;
          Circle c(r);
          cout << "输出面积" << endl;
          cout << c.GetArea() << endl;
          return 0;
      }
      ~~~

7. 编写一个基于对象的程序，求出3个长方柱的体积。数据成员包括length(长)、width(宽)、height(高)。要求用成员函数实现以下功能：

   * 由键盘分别输入3个长方柱的长、宽、高；

   * 计算长方柱的体积；

   * 输出3个长方柱的体积。

     ~~~C++
     #include <iostream>
     using namespace std;
     
     class Cuboid{//定义Cuboid类
         double length;
         double width;
         double height;
         public:
             void Cuboidinput(){
                 cin >> length >> width >> height;
                 return;
             }
             double GetVolume(){
                 return length*width*height;
             }
             void Volumeoutput(){
                 cout << GetVolume() << endl;
                 return;
             }
     };
     
     int main(void){
         Cuboid c1,c2,c3;
         cout << "输入三个长方体的长宽高" << endl;
         c1.Cuboidinput();
         c2.Cuboidinput();
         c3.Cuboidinput();
         cout << "输出三个长方体的体积" << endl;
         c1.Volumeoutput();
         c2.Volumeoutput();
         c3.Volumeoutput();
         return 0;
     }
     ~~~
