title: Leetcode每日一题：145.二叉树的后序遍历
tags: ["Leetcode","C++"]
---

思路：后序遍历超级简单的，就不讲了

~~~c++
class Solution {
public:
    vector<int> ans;
    void dfs(TreeNode* root){
        if(root == nullptr)
            return;
        dfs(root->left);
        dfs(root->right);
        ans.push_back(root->val);
        return;
    }
    vector<int> postorderTraversal(TreeNode* root) {
        dfs(root);
        return ans;
    }
};
~~~
